017package de.dentrassi.asyncapi.gson.time;
018
019import java.io.IOException;
020import java.time.Instant;
021import java.time.ZoneOffset;
022import java.time.ZonedDateTime;
023import java.time.format.DateTimeFormatter;
024import java.time.temporal.ChronoField;
025import java.time.temporal.TemporalAccessor;
026import java.time.temporal.TemporalQuery;
027import java.util.function.Consumer;
028import java.util.function.Function;
029
030import com.google.gson.Gson;
031import com.google.gson.GsonBuilder;
032import com.google.gson.JsonSyntaxException;
033import com.google.gson.TypeAdapter;
034import com.google.gson.TypeAdapterFactory;
035import com.google.gson.reflect.TypeToken;
036import com.google.gson.stream.JsonReader;
037import com.google.gson.stream.JsonToken;
038import com.google.gson.stream.JsonWriter;
039
040public class DateTimeAdapterFactory implements TypeAdapterFactory {
041
042    private static final String ISO_8601_DATE_FORMAT = "yyyy-MM-dd'T'HH:mm:ss.SSS'Z'";
043
044    private static class TemporalAccessorTypeAdapter<T extends TemporalAccessor> extends TypeAdapter<T> {
045
046        private final DateTimeFormatter formatter;
047        private final TemporalQuery<T> query;
048        private final Function<T, ? extends TemporalAccessor> whenWriting;
049
050        public TemporalAccessorTypeAdapter(final DateTimeFormatter formatter, final TemporalQuery<T> query, final Function<T, ? extends TemporalAccessor> whenWriting) {
051            this.formatter = formatter;
052            this.query = query;
053            this.whenWriting = whenWriting;
054        }
055
056        @Override
057        public void write(final JsonWriter out, final T value) throws IOException {
058            if (value == null) {
059                out.nullValue();
060            } else {
061                out.value(this.formatter.format(this.whenWriting.apply(value)));
062            }
063        }
064
065        @Override
066        public T read(final JsonReader in) throws IOException {
067            final JsonToken next = in.peek();
068
069            switch (next) {
070            case STRING: {
071                final String value = in.nextString();
072                return this.formatter.parse(value, this.query);
073            }
074            case NULL:
075                in.nextNull();
076                return null;
077            default:
078                throw new JsonSyntaxException("Invalid content for timestamp");
079            }
080        }
081
082    }
083
084    private final DateTimeFormatter formatter;
085    private final Function<TemporalAccessor, ? extends TemporalAccessor> whenWriting;
086
087    public DateTimeAdapterFactory(final DateTimeFormatter formatter, final Function<TemporalAccessor, ? extends TemporalAccessor> whenWriting) {
088        this.formatter = formatter;
089        this.whenWriting = whenWriting;
090    }
091
092    @Override
093    public <T> TypeAdapter<T> create(final Gson gson, final TypeToken<T> token) {
094
095        final Class<? super T> clazz = token.getRawType();
096
097        if (Instant.class.isAssignableFrom(clazz)) {
098            @SuppressWarnings("unchecked")
099            final TypeAdapter<T> result = (TypeAdapter<T>) new TemporalAccessorTypeAdapter<>(this.formatter, Instant::from, this.whenWriting);
100            return result;
101        } else if (ZonedDateTime.class.isAssignableFrom(clazz)) {
102            @SuppressWarnings("unchecked")
103            final TypeAdapter<T> result = (TypeAdapter<T>) new TemporalAccessorTypeAdapter<>(this.formatter, ZonedDateTime::from, this.whenWriting);
104            return result;
105        }
106
107        return null;
108    }
109
110    public static Consumer<GsonBuilder> iso8601() {
111        return builder -> {
112            builder.registerTypeAdapterFactory(new DateTimeAdapterFactory(DateTimeFormatter.ISO_INSTANT.withZone(ZoneOffset.UTC), value -> value));
113            builder.setDateFormat(ISO_8601_DATE_FORMAT);
114        };
115    }
116
117    public static Consumer<GsonBuilder> iso8601WithTimezone() {
118        return builder -> {
119            builder.registerTypeAdapterFactory(new DateTimeAdapterFactory(DateTimeFormatter.ISO_DATE_TIME, DateTimeAdapterFactory::makeZoned));
120            builder.setDateFormat(ISO_8601_DATE_FORMAT);
121        };
122    }
123
124    public static TemporalAccessor makeZoned(final TemporalAccessor value) {
125        if (value == null) {
126            return null;
127        }
128
129        if (value.isSupported(ChronoField.YEAR)) {
130            return value;
131        }
132
133        if (value instanceof Instant) {
134            return ((Instant) value).atOffset(ZoneOffset.UTC);
135        }
136
137        throw new IllegalArgumentException("Unable to process date time type: " + value.getClass());
138    }
139
140}
